<Tabs syncKey="framework">
  <TabItem label="React" icon="seti:react">

    ## Install the package

    Choose your preferred package manager to install the library:

    <Tabs syncKey="pkg">
      <TabItem label="npm" icon="seti:npm">

        ```bash
        npm install @sankyu/react-circle-flags
        ```

      </TabItem>
      <TabItem label="pnpm" icon="pnpm">

        ```bash
        pnpm add @sankyu/react-circle-flags
        ```

      </TabItem>
      <TabItem label="yarn" icon="seti:yarn">

        ```bash
        yarn add @sankyu/react-circle-flags
        ```

      </TabItem>
      <TabItem label="bun" icon="bun">

        ```bash
        bun add @sankyu/react-circle-flags
        ```

      </TabItem>
    </Tabs>

    :::note[Peer Dependency]
    The package expects **React** to already be installed in your project.
    :::

  </TabItem>
